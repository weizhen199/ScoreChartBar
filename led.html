jQuery实用LED样式计时器|倒计数器插件  

Current Time

$('.led2').ClassyLED({
    type: 'time',
    format: 'hh:mm',
    color: '#ed1978',
    backgroundColor: '#cccccc',
    size: 14,
    rounded: 4,
    fontType: 1
});

Countdown to Christmas (days-hours-minutes-seconds)

$('.led3').ClassyLED({
    type: 'countdown',
    format: 'ddd:hh:mm:ss',
    countTo: '2016:12:30',
    color: '#ff0000',
    backgroundColor: '#ffffff',
    size: 10,
    fontType: 2
});

Countdown to New Year (days-hours)

$('.led4').ClassyLED({
    type: 'countdown',
    format: 'ddd:hh',
    color: '#af0',
    backgroundColor: '#101a10',
    size: 10,
    rounded: 6
});

Random numbers

$('.led5').ClassyLED({
    type: 'random',
    color: '#ed1978',
    backgroundColor: '#ffffff',
    size: 17,
    rounded: 9,
    fontType: 3
});
